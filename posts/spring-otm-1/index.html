@Entity
class Class(
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    var id: Long,
    @OneToMany
    var students: MutableSet<Student> = mutableSetOf()
)

OneToMany관계는 기본적으로 JoinTable 기반으로 동작하며 생성되는 Table은 위와 같다.

MappedBy

@Entity
class Student(
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    var id: Long,
    var name: String,
    var classId: Long
    or
    @ManyToOne
    var `class`: Class
)

@Entity
class Class(
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    var id: Long,
    @OneToMany(mappedBy = "classId")
    var students: MutableSet<Student> = mutableSetOf()
    or
    @OneToMany(mappedBy = "class")
    var students: MutableSet<Student> = mutableSetOf()
)
